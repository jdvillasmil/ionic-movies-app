    ```typescript
    import { user } from '@angular/fire/auth';
    export const authGuard: CanActivateFn = () => {
      const auth = inject(Auth);
      const router = inject(Router);
      return user(auth).pipe(
        take(1),
        map((u) => (u ? true : router.createUrlTree(['/login'])))
      );
    };
    ```

    Initials derivation (up to 3 letters):
    ```

    ```

    Color derivation (deterministic HSL hash — no randomness):
    ```
    function hashColor(str: string): string {
      let hash = 0;
      for (let i = 0; i < str.length; i++) {
        hash = str.charCodeAt(i) + ((hash << 5) - hash);
      }
      return `hsl(${Math.abs(hash) % 360}, 50%, 45%)`;
    }
    ```

    Style the initials circle inline: width/height = size px, border-radius 50%, background = hashColor(displayName), color white, display flex, align-items center, justify-content center, font-size = size*0.4 px, font-weight 600, user-select none.
